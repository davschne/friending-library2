Schemas
-------

User
----
uID          : varchar PRIMARY KEY
displayName  : varchar

Friend
------
uID1 : varchar references User(uID)
uID2 : varchar references User(uID)

Book
----
ISBN          : integer PRIMARY KEY
title         : varchar
subtitle      : varchar
publisher     : varchar
publishedDate : integer
description   : varchar
pageCount     : integer
language      : varchar
imageLink     : varchar

Author
------
aID  : serial PRIMARY KEY
name : varchar

Wrote
-----
aID  : integer references Author(aID)
ISBN : integer references Book(ISBN)

Copy
----
cID  : integer PRIMARY KEY
ISBN : integer references Book(ISBN)
uID  : integer references User(uID) - owner
